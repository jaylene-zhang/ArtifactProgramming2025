(* Q1 : String to Characters to String                  *)
(*--------------------------------------------------------------*)

(* 1.1 Turn a string into a list of characters. *)
let string_explode (s : string) : char list =
  List.init (String.length s) (fun i -> String.get s i)

(* 1.2 Turn a list of characters into a string. *)
let string_implode (l : char list) : string =
  List.fold_right (fun c acc -> acc ^ Char.escaped c) l ""

let open_account (pass: password) : bank_account =
  let balance = ref 0 in
  let validate_password p =
    if p <> pass then raise wrong_pass
  in
  {
    update_pass = (fun old_pass new_pass ->
      validate_password old_pass;
      pass := new_pass);
    retrieve = (fun p amount ->
      validate_password p;
      if amount < 0 then raise negative_amount;
      if !balance < amount then raise not_enough_balance;
      balance := !balance - amount);
    deposit = (fun p amount ->
      validate_password p;
      if amount < 0 then raise negative_amount;
      balance := !balance + amount);
    show_balance = (fun p ->
      validate_password p;
      !balance)
  }
