(* Q1 : String to Characters to String                          *)
(*--------------------------------------------------------------*)

(* 1.1 Turn a string into a list of characters. *)
let string_explode (s : string) : char list =
  let len = String.length s in
  List.init len (String.get s)

(* 1.2 Turn a list of characters into a string. *)
let string_implode (l : char list) : string =
  List.fold_left (fun acc c -> acc ^ Char.escaped c) "" l

(*--------------------------------------------------------------*)
(* Q2 : Bank Account                                            *)
(*--------------------------------------------------------------*)

let open_account (pass: password) : bank_account =
  let current_password = ref pass in
  let balance = ref 0 in
  {
    update_pass = (fun old_pass new_pass ->
      if old_pass <> !current_password then raise wrong_pass else
      current_password := new_pass
    );

    deposit = (fun provided_pass amount ->
      if provided_pass <> !current_password then raise wrong_pass else
      if amount < 0 then raise negative_amount else
      balance := !balance + amount
    );

    retrieve = (fun provided_pass amount ->
      if provided_pass <> !current_password then raise wrong_pass else
      if amount < 0 then raise negative_amount else
      if amount > !balance then raise not_enough_balance else
      balance := !balance - amount
    );

    show_balance = (fun provided_pass ->
      if provided_pass <> !current_password then raise wrong_pass else
      !balance
    )
  }
