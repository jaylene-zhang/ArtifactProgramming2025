(* Q1 : String to Characters to String *)
let string_explode (s : string) : char list =
  let n = String.length s in
  List.init n (fun i -> String.get s i)

let string_implode (l : char list) : string =
  List.fold_right (fun c s -> String.make 1 c ^ s) l ""

(* Q2 : Bank Account *)
let open_account (pass: password) : bank_account =
  let password = ref pass in
  let balance = ref 0 in
  {
    update_pass = (fun old new_ ->
      if old <> !password then raise wrong_pass;
      password := new_
    );
    retrieve = (fun pw amount ->
      if pw <> !password then raise wrong_pass;
      if amount < 0 then raise negative_amount;
      if !balance < amount then raise not_enough_balance;
      balance := !balance - amount
    );
    deposit = (fun pw amount ->
      if pw <> !password then raise wrong_pass;
      if amount < 0 then raise negative_amount;
      balance := !balance + amount
    );
    show_balance = (fun pw ->
      if pw <> !password then raise wrong_pass;
      !balance
    );
  }
