let rec traverse (f : 'a -> 'b option) (l : 'a list) : 'b list option =
  match l with
  | [] -> Some []
  | hd :: tl ->
      match f hd with
      | None -> None
      | Some hd' -> (
          match traverse f tl with
          | None -> None
          | Some tl' -> Some (hd' :: tl')
      )

let seq2 (m1 : 'a1 Maybe.t) (m2 : 'a2 Maybe.t) : ('a1 * 'a2) Maybe.t =
  let open Maybe in
  match m1, m2 with
  | Just x1, Just x2 -> Just (x1, x2)
  | _ -> Nothing
