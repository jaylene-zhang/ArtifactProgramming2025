let rec traverse (f : 'a -> 'b option) (l : 'a list) : 'b list option =
  match l with
  | [] -> Some []
  | h :: t ->
      match f h with
      | None -> None
      | Some hd ->
          (match traverse f t with
           | None -> None
           | Some tl -> Some (hd :: tl))
