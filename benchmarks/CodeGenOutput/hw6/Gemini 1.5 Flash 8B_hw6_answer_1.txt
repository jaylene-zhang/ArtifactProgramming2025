let rec traverse (f : 'a -> 'b option) (l : 'a list) : 'b list option =
  match l with
  | [] -> Some []
  | h :: t ->
    match f h with
    | None -> None
    | Some v ->
      (match traverse f t with
       | None -> None
       | Some tl -> Some (v :: tl))

  match l with
  | [] -> CList (fun f _ -> f ())
