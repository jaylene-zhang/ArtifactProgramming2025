(* Q1: Corrected implementation of distance *)
let distance (x1, y1) (x2, y2) =
  abs (y2 - y1) + abs (x2 - x1)

(* Q2: Implementation of binomial using a helper factorial function *)
let binomial n k =
  let rec factorial x =
    if x = 0 then 1 else x * factorial (x - 1)
  in
  factorial n / (factorial k * factorial (n - k))

(* Q3: Implementation of tail-recursive Lucas function *)
let rec lucas_helper n a b =
  if n = 0 then a
  else if n = 1 then b
  else lucas_helper (n - 1) b (a + b)

let lucas n =
  lucas_helper n 2 1
