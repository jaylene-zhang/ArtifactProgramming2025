(* Q1: Correct implementation of Manhattan distance *)
let distance (x1, y1) (x2, y2) = 
  (abs (x2 - x1)) + (abs (y2 - y1))


(* Q2: Implementation of binomial coefficient using factorials *)
let binomial n k =
  let rec factorial m =
    if m = 0 then 1
    else m * factorial (m-1)
  in
  (factorial n) / ((factorial k) * (factorial (n-k)))


(* Q3: Tail-recursive implementation of Lucas numbers *)
let rec lucas_helper n a b =
  if n = 0 then a
  else lucas_helper (n-1) b (a+b)

let lucas n =
  lucas_helper n 2 1
