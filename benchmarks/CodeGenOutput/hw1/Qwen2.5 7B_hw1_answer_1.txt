let rec factorial =
  let rec fact_aux acc n =
    if n = 0 then acc
    else fact_aux (acc * n) (n - 1)
  in fact_aux 1

let binomial n k =
  if k > n || k < 0 then 0 else (factorial n) / ((factorial k) * (factorial (n - k)))
