(* Q1 *)
let distance (x1, y1) (x2, y2) =
  abs (y2 - y1) + abs (x2 - x1)

(* Q2 *)
let binomial n k =
  let rec factorial x =
    if x = 0 then 1
    else x * factorial (x - 1)
  in
  factorial n / (factorial k * factorial (n - k))

(* Q3 *)
let rec lucas_helper n l1 l2 =
  if n = 0 then l1
  else lucas_helper (n - 1) l2 (l1 + l2)

let lucas n =
  lucas_helper n 2 1
