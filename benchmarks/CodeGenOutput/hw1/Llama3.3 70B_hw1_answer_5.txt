let distance (x1, y1) (x2, y2) =

(* Q2 *)
let binomial n k =
  let rec factorial i =
    if i = 0 then 1
    else i * factorial (i - 1) in
  if k < 0 || k > n then
    failwith "Invalid input"
  else
    factorial n / (factorial k * factorial (n - k))

let lucas n =
  let rec lucas_helper a b n =
    if n = 0 then a
    else lucas_helper b (a + b) (n - 1) in
  lucas_helper 2 1 n
