(* Q1. Corrected implementation for Manhattan distance *)
let distance (x1, y1) (x2, y2) =
  abs (x2 - x1) + abs (y2 - y1)

(* Q2. Implementation for binomial coefficient using factorial *)
let binomial n k =
  let rec factorial x =
    if x = 0 then 1 else x * factorial (x - 1)
  in
  factorial n / (factorial k * factorial (n - k))

(* Q3. Tail-recursive implementation of Lucas numbers *)
let rec lucas_helper n l0 l1 =
  if n = 0 then l0
  else if n = 1 then l1
  else lucas_helper (n - 1) l1 (l0 + l1)

let lucas n =
  lucas_helper n 2 1
