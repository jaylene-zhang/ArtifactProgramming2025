let distance (x1, y1) (x2, y2) =
  abs (x2 - x1) + abs (y2 - y1)

let binomial n k =
  let rec factorial x =
    if x <= 1 then 1 else x * factorial (x - 1)
  in
  factorial n / (factorial k * factorial (n - k))

let rec lucas_helper n l_n2 l_n1 =
  if n = 0 then l_n2
  else if n = 1 then l_n1
  else lucas_helper (n - 1) l_n1 (l_n2 + l_n1)

let lucas n =
  lucas_helper n 2 1
