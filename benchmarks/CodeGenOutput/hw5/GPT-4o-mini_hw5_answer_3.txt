(* Question 1: Tree Depth *)

let rec tree_depth_cps t =
  let rec depth_helper t sc =
    match t with
    | Empty -> sc 0
    | Tree (l, _, r) ->
        depth_helper l (fun l_depth ->
          depth_helper r (fun r_depth ->
            sc (1 + max l_depth r_depth)))
  in
  depth_helper t (fun depth -> depth)

(* Question 2: Finding Subtrees *)

let find_subtree_cps ls tree =
  let rec helper ls tree sc fc =
    match ls, tree with
    | [], _ -> sc (Some tree)
    | _, Empty -> fc ()
    | h::t, Tree (l, v, r) when h = v ->
        helper t l (fun l_res ->
          match l_res with 
          | Some _ as r -> sc r 
          | None -> helper t r sc fc)
    | _ -> fc ()
  in
  helper ls tree (fun res -> Some res) (fun () -> None)
